# 多平台内容解析思路与实现指南

## 🎯 核心解析思路

所有平台的内容解析都遵循相同的**网页数据挖掘**模式：

```
分享链接 → 重定向处理 → 获取页面HTML → 提取JavaScript数据 → 解析结构化信息
```

### 通用流程步骤

1. **🔗 链接处理**：处理分享链接的重定向，获取真实页面URL
2. **📄 页面获取**：使用合适的User-Agent获取完整HTML页面
3. **🔍 数据提取**：从页面中的JavaScript变量提取JSON数据
4. **📝 信息解析**：将原始数据解析为统一的结构化格式

### 2. 解析方法结构
```javascript
async parsePlatformUrl(shareUrl) {
  console.log("开始解析[平台]链接:", shareUrl)
  
  // 第一步：获取重定向后的真实链接
  const shareResp = await axios.get(shareUrl, { 
    headers: getRandomUA(),
    maxRedirects: 5,
    timeout: 10000
  })
  const realUrl = shareResp.request.res.responseUrl
  console.log("重定向后的真实链接:", realUrl)

  // 第二步：提取内容ID（根据平台URL格式调整正则）
  const idMatch = realUrl.match(/\/path\/([a-zA-Z0-9]+)/)
  if (!idMatch) {
    throw new Error('无法从链接中提取内容ID')
  }
  const contentId = idMatch[1]
  console.log("提取到的内容ID:", contentId)

  // 第三步：获取页面内容并解析数据
  const pageResp = await axios.get(realUrl, { 
    headers: {
      ...getRandomUA(),
      'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8',
      'Accept-Language': 'zh-CN,zh;q=0.8,zh-TW;q=0.7,zh-HK;q=0.5,en-US;q=0.3,en;q=0.2'
    }
  })

  const html = pageResp.data
  
  // 第四步：提取JavaScript数据（根据平台调整数据源）
  const scriptMatch = html.match(/window\.__DATA_SOURCE__\s*=\s*({.*?})\s*<\/script>/s)
  if (!scriptMatch) {
    throw new Error('无法从页面中提取数据')
  }

  // 第五步：解析JSON数据
  let jsonData
  try {
    let jsonStr = scriptMatch[1]
    jsonStr = jsonStr.replace(/undefined/g, 'null') // 清理常见问题
    jsonData = JSON.parse(jsonStr)
  } catch (parseError) {
    console.error('JSON解析失败:', parseError.message)
    throw new Error('页面数据解析失败')
  }

  // 第六步：定位具体数据（根据平台数据结构调整）
  const contentData = jsonData.path?.to?.data?.[contentId]
  if (!contentData) {
    throw new Error('未找到内容数据')
  }

  console.log("获取到内容数据:", contentData.title)

  // 第七步：解析为统一格式
  return this.parsePlatformData(contentData, contentId)
}
```

### 3. 数据解析方法
```javascript
parsePlatformData(rawData, contentId) {
  const title = rawData.title || `platform_${contentId}`
  const cleanTitle = title.replace(/[\\/:*?"<>|]/g, '_')
  
  // 判断内容类型
  const isVideo = rawData.type === 'video' || rawData.videoUrl
  
  let videoUrl = null
  let coverImg = []
  let allImg = []

  if (isVideo) {
    // 视频处理逻辑
    videoUrl = rawData.videoUrl
    if (rawData.cover) {
      coverImg = [rawData.cover]
    }
  } else {
    // 图片处理逻辑
    if (rawData.images && rawData.images.length > 0) {
      allImg = rawData.images.map(img => img.url).filter(Boolean)
    }
  }

  // 作者信息
  const author = {
    nickname: rawData.author?.name || '未知用户',
    user_id: rawData.author?.id || '',
    avatar: rawData.author?.avatar || ''
  }

  // 统计信息
  const statistics = {
    liked_count: parseInt(rawData.stats?.likes) || 0,
    collected_count: parseInt(rawData.stats?.collects) || 0,
    comment_count: parseInt(rawData.stats?.comments) || 0,
    share_count: parseInt(rawData.stats?.shares) || 0
  }

  return {
    // 内容信息
    content_id: contentId,
    type: isVideo ? 'video' : 'image',
    title: cleanTitle,
    desc: rawData.description || '',
    cover: coverImg,
    allImg: allImg,
    url: videoUrl,
    // 作者信息
    author: author,
    // 统计信息
    statistics: statistics,
    // 标签信息
    tags: rawData.tags ? rawData.tags.map(tag => tag.name || tag) : [],
    // 平台标识
    platform: 'platform'
  }
}
```

### 2. 数据结构分析
```javascript
// 递归查找目标数据
function findContentData(obj, targetId, path = '') {
  if (typeof obj !== 'object' || obj === null) return
  
  for (const [key, value] of Object.entries(obj)) {
    const currentPath = path ? `${path}.${key}` : key
    
    // 查找包含目标ID的数据
    if (key === targetId || (typeof value === 'object' && 
        JSON.stringify(value).includes(targetId))) {
      console.log(`找到相关数据 - 路径: ${currentPath}`)
      console.log('数据类型:', Array.isArray(value) ? 'Array' : typeof value)
      if (typeof value === 'object') {
        console.log('对象键:', Object.keys(value))
      }
    }
    
    // 递归查找
    if (typeof value === 'object' && path.split('.').length < 3) {
      findContentData(value, targetId, currentPath)
    }
  }
}
```
